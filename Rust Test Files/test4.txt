fn main(){
    // simple else if ladder
    // output a string based on a given number
    let mut n = 10;
    let mut str = "";

    if n < 10 {
        str = "The number is less than 10";
    }
    else if n == 10{
        str = "The number is equal to 10";
    }
    else {
        str = "The number is greater than 10";
    }
}
